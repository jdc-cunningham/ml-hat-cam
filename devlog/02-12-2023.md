```
near_vars = [
  8,
  10,
  12,
  14,
  16,
  14,
  12
]

far_vars = [
  8,
  6,
  4,
  2,
  0
]

prev_var = 0
next_var = 0
max_var = 0
dir_near = None

def focus_near(i):
  var = near_vars[i]
  return var

def focus_far(i):
  var = far_vars[i]
  return var

for i in range (0, 7, 1): # frame loop
  # get first sample
  if (prev_var == 0):
    prev_var = focus_near(i)
    max_var = prev_var
    continue

  # get second sample
  if (next_var == 0):
    next_var = focus_near(i)

    if (next_var > max_var):
      max_var = next_var
      continue
  else: # decide direction to keep going
    if (next_var > prev_var):
      dir_near = True
    else:
      dir_near = False

  if (dir_near):
    cur_var = focus_near(i)
  else:
    cur_var = focus_far(i)
  
  # find max value and stop
  if (cur_var):
    if (cur_var > max_var):
      max_var = cur_var
    elif (cur_var <= max_var):
      break
```
